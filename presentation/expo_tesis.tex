% ── FRAME 2: MOTIVACIÓN ───────────────────────────────────────────────────
\begin{frame}{Motivación}
  \begin{itemize}
    \setlength\itemsep{14pt}
    \item Yape y Plin superaron \textbf{105 millones de operaciones mensuales}
          en enero de 2024 (BCRP), pero la adopción es profundamente desigual:
          los adultos mayores en pobreza extrema son el grupo con
          \textbf{menor adopción digital}.

    \item El Estado peruano \textbf{digitaliza transferencias}
          (Pensión 65, Juntos, Beca 18) como mecanismo de inclusión
          financiera, asumiendo que la recepción del bono activa el uso
          de billeteras.

    \item \textbf{No existe evidencia causal} sobre si esa activación
          ocurre efectivamente entre los adultos mayores beneficiarios de
          Pensión 65 en Perú.
  \end{itemize}
\end{frame}

% ── FRAME 3: PREGUNTA DE INVESTIGACIÓN ────────────────────────────────────
\begin{frame}{Pregunta de Investigación}
  \vspace{6pt}
  \begin{block}{}
    \centering
    \large
    \textbf{¿Cruzar el umbral de elegibilidad de Pensión 65 a los 65 años
    aumenta la adopción de billeteras digitales\\
    entre adultos mayores en el Perú?}
  \end{block}

  \vspace{14pt}
  \begin{itemize}
    \setlength\itemsep{10pt}
    \item \textbf{Primera evidencia causal para Perú} usando ENAHO 2024
          (N\,=\,113{,}755 individuos) y diseño de regresión
          discontinua difusa (fuzzy RDD) en el umbral de edad 65.

    \item \textbf{Primera encuesta que mide billeteras digitales}
          como categoría diferenciada (Yape, Plin, Cuenta DNI),
          tanto tenencia como uso activo.
  \end{itemize}
\end{frame}

% ── FRAME 6: ESTRATEGIA DE IDENTIFICACIÓN ─────────────────────────────────
\begin{frame}{Estrategia de Identificación: Fuzzy RDD en el Umbral de 65 Años}
  \small
  \textit{Comparamos personas justo antes y después del umbral de edad: son
  comparables en todo salvo su elegibilidad administrativa al programa.}

  \vspace{4pt}
  \begin{itemize}
    \setlength\itemsep{3pt}
    \item \textbf{Variable de corrida:} edad centrada en 65 ($r_i = \text{Edad}_i - 65$)
    \item \textbf{Umbral administrativamente rígido,} verificado por DNI,
          no manipulable por el individuo
    \item \textbf{Diseño \emph{fuzzy}:} Pensión 65 requiere también clasificación
          SISFOH extremo pobre $\Rightarrow$ no todos los mayores de 65 reciben
  \end{itemize}

  \vspace{6pt}
  \textbf{Ecuación 1 — Primera Etapa:}
  \begin{equation*}
    D_i = \alpha_0
        + \rho\,\mathbf{1}[\text{Edad}_i \geq 65]
        + \alpha_1 r_i
        + \alpha_2\,r_i \cdot \mathbf{1}[r_i \geq 0]
        + \varepsilon_i
    \quad\bigl|\;|r_i| \leq h\bigr.
  \end{equation*}

  \textbf{Ecuación 2 — Segunda Etapa (IV):}
  \begin{equation*}
    Y_i = \beta_0
        + \tau\,\widehat{D}_i
        + \beta_1 r_i
        + \beta_2\,r_i \cdot \mathbf{1}[r_i \geq 0]
        + u_i
    \quad\bigl|\;|r_i| \leq h\bigr.
  \end{equation*}

  \textbf{Ecuación 3 — Relación ITT–LATE (Angrist \& Pischke, 2009):}
  \begin{equation*}
    \hat{\tau}_{\text{LATE}}
    = \frac{\hat{\tau}_{\text{RF}}}{\hat{\rho}}
    = \frac{%
        \lim_{r \downarrow 0}\mathbb{E}[Y_i \mid r_i=r]
        - \lim_{r \uparrow 0}\mathbb{E}[Y_i \mid r_i=r]}{%
        \lim_{r \downarrow 0}\mathbb{E}[D_i \mid r_i=r]
        - \lim_{r \uparrow 0}\mathbb{E}[D_i \mid r_i=r]}
  \end{equation*}
\end{frame}

% ── FRAME 7: HALLAZGO CRÍTICO ─────────────────────────────────────────────
\begin{frame}{Hallazgo Crítico: Un Problema en el Diseño Original}
  \textit{La variable POBREZA monetaria del INEI \textbf{no equivale} al
  criterio SISFOH que usa Pensión 65 para determinar elegibilidad.}

  \vspace{8pt}
  \begin{center}
  \small
  \begin{tabular}{l rr}
    \toprule
    \textbf{Clasificación INEI (POBREZA)} &
    \textbf{Receptores reales P65} & \textbf{\%} \\
    \midrule
    POBREZA\,=\,1\ \ (extremo pobre monetario)
      & \ \ \ 501 & 11.3\% \\
    POBREZA\,=\,2\ \ (pobre no extremo monetario)
      & 1{,}118 & 25.2\% \\
    POBREZA\,=\,3\ \ (no pobre monetario)
      & 2{,}818 & 63.5\% \\
    \midrule
    \textit{Total}
      & 4{,}437 & 100.0\% \\
    \bottomrule
  \end{tabular}
  \end{center}

  \vspace{6pt}
  \begin{alertblock}{}
    \textbf{Restringir a POBREZA\,=\,1 excluye al 88.7\% de los receptores
    reales.}
  \end{alertblock}

  \vspace{4pt}
  \textbf{Tres causas:}
  \begin{itemize}
    \setlength\itemsep{2pt}
    \item \textit{Causalidad inversa:} la transferencia eleva el gasto
          del hogar $\Rightarrow$ el receptor «sube» de POBREZA\,=\,1 a 2 o 3
    \item \textit{SISFOH mide activos y condiciones de vivienda,}
          no gasto corriente (proxy means test de MIDIS)
    \item \textit{Error de clasificación} entre ambos sistemas
  \end{itemize}
\end{frame}

% ── FRAME 13: PRUEBAS DE VALIDEZ ──────────────────────────────────────────
\begin{frame}{Pruebas de Validez del Diseño}
  \begin{itemize}
    \setlength\itemsep{12pt}
    \item[\pcheck] \textbf{Test de densidad (McCrary):}
          $T = 2.00$ — sin evidencia de manipulación
          del running variable alrededor del umbral de 65 años.

    \item[\pcheck] \textbf{Balance de covariables:}
          cinco variables predeterminadas (internet, smartphone,
          ingreso pc, educación, área) sin discontinuidad
          estadísticamente significativa en el umbral.

    \item[\pcheck] \textbf{Sensibilidad al ancho de banda:}
          estimaciones estables para $h \in \{7, 10, 14, 18, 24\}$ años.

    \item[\pcheck] \textbf{Especificación cuadrática:}
          resultados cualitativamente idénticos al baseline lineal.

    \item[\pcheck] \textbf{Placebo Juntos}
          ($\hat{\tau} = 0.010$, IC$_{95\%} = [-0.002,\,+0.022]$):
          no hay discontinuidad espuria en transferencias sin
          corte de edad a los 65.
  \end{itemize}
\end{frame}

% ── FRAME 14: INTERPRETACIÓN DEL NULO ────────────────────────────────────
\begin{frame}{¿Por Qué el Resultado es Nulo?}
  \begin{itemize}
    \setlength\itemsep{14pt}
    \item \textbf{Cuentas durmientes}
          \textit{(Bachas et al., 2018):}
          los beneficiarios reciben la transferencia pero retiran efectivo
          en ventanilla sin adoptar la billetera activamente.
          La apertura de cuenta no equivale al uso.

    \item \textbf{Gradiente de edad en tecnología}
          \textit{(Czaja et al., 2006):}
          los adultos mayores enfrentan costos de aprendizaje
          más altos que anulan el incentivo institucional del bono.

    \item \textbf{Canal institucional insuficiente}
          \textit{(Banerjee et al., 2020):}
          el acceso al sistema financiero es necesario pero
          no suficiente sin \textbf{alfabetización digital},
          acceso subsidiado a smartphones y
          \textbf{acompañamiento presencial}.
  \end{itemize}
\end{frame}

        \footnotesize
        \begin{itemize}
          \setlength\itemsep{4pt}
          \item Construir proxy SISFOH con ACP usando variables de

  \end{columns}

% ── FRAME 16: CONCLUSIONES ────────────────────────────────────────────────
\begin{frame}{Conclusiones}
  \begin{itemize}
    \setlength\itemsep{14pt}
    \item Cruzar el umbral de elegibilidad de Pensión 65
          \textbf{no genera un aumento estadísticamente significativo}
          en la adopción de billeteras digitales
          ($\hat{\tau}_{\text{ITT}} = -0.002$,
           IC$_{95\%} = [-0.040,\,+0.036]$),
          estable en todas las especificaciones y anchos de banda.

    \item El nulo \textbf{no es solo dilución estadística:}
          con el diseño fuzzy RDD corregido,
          la primera etapa es fuerte ($F = 310$) y la
          forma reducida es genuinamente pequeña
          ($\hat{\tau}_{\text{RF}} = -0.001$) —
          el efecto es pequeño en la población general.

    \item La \textbf{digitalización institucional de transferencias
          no es suficiente por sí sola:}
          se necesitan complementos de
          \textbf{alfabetización digital},
          acceso subsidiado a smartphones y
          \textbf{acompañamiento presencial}
          para lograr inclusión financiera activa.
  \end{itemize}
\end{frame}
